Spring boot oauth2

https://github.com/royclarkson/spring-rest-service-oauth

mvn clean package
mvn spring-boot:run

gradle build
gradle bootRun

java -jar target/spring-boot-oauth-0.1.0.jar

curl http://localhost:8080/index/test

curl http://localhost:8080/employee/list

curl -X POST -kliu \
    clientapp:123456 \
    http://localhost:8080/oauth/token \
    -H "Accept: application/json" \
    -d grant_type=password \
    -d scope=read%20write \
    -d client_id=clientapp \
    -d client_secret=123456 \
    -d username=user \
    -d password=user

curl -X POST -liu \
    clientapp:123456 \
    http://localhost:8080/oauth/token \
    -H "Accept: application/json" \
    -d "password=user&username=user&grant_type=password&scope=read%20write&client_secret=123456&client_id=clientapp"

curl http://localhost:8080/employee/list -H "Authorization: Bearer 215daf9f-9228-4d28-b286-cf80a5763dd5"
